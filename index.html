async function loadSLV() {
try {
const res = await fetch("https://www.ishares.com/us/products/239855/slv/1467271812596.ajax?fileType=csv");
const text = await res.text();

// Parse CSV lines
const rows = text.split("\n").slice(1);
const dates = [];
const holdings = [];

rows.forEach(row => {
const cols = row.split(",");
if (cols.length > 5) {
dates.push(cols[0]); // date
holdings.push(parseFloat(cols[5])); // holdings (in ounces)
}
});

// Draw chart
new Chart(document.getElementById("slvChart"), {
type: "line",
data: {
labels: dates.reverse().slice(-60), // latest 60 days
datasets: [{
data: holdings.reverse().slice(-60),
borderWidth: 2
}]
},
options: {
responsive: true,
scales: { y: { beginAtZero: false } }
}
});

} catch (err) {
console.log(err);
}
}

loadSLV();
